            CreateAdmin(AdminCreateDto dto)

        (bool IsSuccess, string Message)

[v] 實作 AdminService
    - 位置：ISpanShop.Services/AdminService.cs

    - CreateAdmin 邏輯：
        1. 確認 AdminLevelId 不為 1（不可直接新增超級管理員）
        2. 取得流水號
           int seq = _adminRepository.GetNextAdminSequence()
        3. 組成帳號與 Email
           string account = $"ADM{seq:D3}"
           string email   = $"{account}@ispan.com"
        4. 確認帳號不重複（IsAccountExists）
        5. 角色固定為 Admin
           int roleId = 1   
        6. Hash 初始密碼
           string passwordHash = BCrypt.HashPassword(dto.Password)
        7. 呼叫 Repository.CreateAdmin(
               account, email, passwordHash, roleId, dto.AdminLevelId)
        8. 回傳 GeneratedAccount 供畫面顯示

    - DeactivateAdmin 邏輯：
        1. userId != currentUserId（不可停用自己）
        2. 若目標是超級管理員，檢查超級管理員數量 > 1
           SELECT COUNT(1) FROM Users U
           JOIN AdminLevels AL ON U.AdminLevelId = AL.Id
           WHERE AL.Id = 1 AND U.IsBlacklisted = 0
        3. 呼叫 Repository.DeactivateAdmin(userId)

    - VerifyLogin 邏輯：
        1. 呼叫 GetAdminByAccount(account)
        2. 帳號不存在 -> 回傳 null
        3. BCrypt.Verify(password, admin.PasswordHash)
        4. 驗證成功 -> 回傳 AdminDto（含 IsFirstLogin）
        5. 驗證失敗 -> 回傳 null

    - ChangePassword 邏輯：
        1. NewPassword == ConfirmPassword
        2. 新密碼至少 8 碼且含英文與數字
        3. Hash 新密碼
        4. 呼叫 Repository.ChangePassword()
        5. 呼叫 Repository.SetFirstLoginComplete()

[V] 實作 StoreRepository（ADO.NET）
    - 位置：ISpanShop.Repositories/Stores/StoreRepository.cs

    - SQL GetAllStores：
        SELECT S.Id            AS StoreId,
               S.UserId,
               U.Account       AS OwnerAccount,
               S.StoreName,
               S.Description,
               S.IsVerified,
               U.IsBlacklisted,
               S.CreatedAt,
               COUNT(P.Id)     AS ProductCount
        FROM   Stores S
        JOIN   Users U   ON S.UserId = U.Id
        LEFT JOIN Products P ON P.StoreId = S.Id
                             AND P.IsDeleted = 0
        GROUP BY S.Id, S.UserId, U.Account, S.StoreName,
                 S.Description, S.IsVerified,
                 U.IsBlacklisted, S.CreatedAt
        ORDER BY S.CreatedAt DESC

    - SQL GetStoreById：
        SELECT S.Id              AS StoreId,
               S.UserId,
               U.Account         AS OwnerAccount,
               S.StoreName,
               S.Description,
               S.IsVerified,
               U.IsBlacklisted,
               S.CreatedAt,
               COUNT(P.Id)       AS ProductCount,
               SUM(CASE WHEN P.Status = 1
                        THEN 1 ELSE 0 END)   AS ActiveProductCount,
               ISNULL(SUM(P.TotalSales), 0)  AS TotalSales
        FROM   Stores S
        JOIN   Users U   ON S.UserId = U.Id
        LEFT JOIN Products P ON P.StoreId = S.Id
                             AND P.IsDeleted = 0
        WHERE  S.Id = @StoreId
        GROUP BY S.Id, S.UserId, U.Account, S.StoreName,
                 S.Description, S.IsVerified,
                 U.IsBlacklisted, S.CreatedAt

    - SQL ToggleVerified：
        UPDATE Stores
        SET    IsVerified = @IsVerified
        WHERE  Id = @StoreId

    - SQL ToggleBlacklist：
        UPDATE Users
        SET    IsBlacklisted = @IsBlacklisted,
               UpdatedAt = GETDATE()
        WHERE  Id = @UserId

        -- 基底查詢加上 WHERE 條件與 ORDER BY
        SELECT S.Id            AS StoreId,
               S.UserId,
               U.Account       AS OwnerAccount,
               S.StoreName,
               S.Description,
               S.IsVerified,
               S.IsBlacklisted,
               S.CreatedAt,
               COUNT(P.Id)     AS ProductCount
        FROM   Stores S
        JOIN   Users U   ON S.UserId = U.Id
        LEFT JOIN Products P ON P.StoreId = S.Id
                             AND P.IsDeleted = 0
        WHERE  1 = 1
          AND  (@Keyword IS NULL
                OR S.StoreName LIKE '%' + @Keyword + '%'
                OR U.Account   LIKE '%' + @Keyword + '%')
          AND  (@VerifyStatus = 'all'
                OR (@VerifyStatus = 'verified'   AND S.IsVerified = 1)
                OR (@VerifyStatus = 'unverified' AND S.IsVerified = 0))
          AND  (@BlockStatus = 'all'
                OR (@BlockStatus = 'normal'  AND S.IsBlacklisted = 0)
                OR (@BlockStatus = 'blocked' AND S.IsBlacklisted = 1))
        GROUP BY S.Id, S.UserId, U.Account, S.StoreName,
                 S.Description, S.IsVerified,
                 S.IsBlacklisted, S.CreatedAt
        ORDER BY
            CASE WHEN @SortColumn = 'StoreName'  AND @SortDir = 'asc'
                 THEN S.StoreName END ASC,
            CASE WHEN @SortColumn = 'StoreName'  AND @SortDir = 'desc'
                 THEN S.StoreName END DESC,
            CASE WHEN @SortColumn = 'ProductCount' AND @SortDir = 'asc'
                 THEN COUNT(P.Id) END ASC,
            CASE WHEN @SortColumn = 'ProductCount' AND @SortDir = 'desc'
                 THEN COUNT(P.Id) END DESC,
            CASE WHEN @SortColumn = 'CreatedAt'  AND @SortDir = 'asc'
                 THEN S.CreatedAt END ASC,
            CASE WHEN @SortColumn = 'CreatedAt'  AND @SortDir = 'desc'
                 THEN S.CreatedAt END DESC
        OFFSET @Offset ROWS FETCH NEXT @PageSize ROWS ONLY;

        -- 同條件 COUNT（取得總筆數）
        SELECT COUNT(DISTINCT S.Id)
        FROM   Stores S
        JOIN   Users U ON S.UserId = U.Id
        WHERE  ... （同上 WHERE 條件）

[V] 新增 RequirePermissionAttribute
    - 位置：ISpanShop.MVC/Middleware/RequirePermissionAttribute.cs
    - 內容：
 
        using ISpanShop.Common.Helpers;
 
        public class RequirePermissionAttribute
            : Attribute, IAuthorizationFilter
        {
            private readonly string _permissionKey;
 
            public RequirePermissionAttribute(string permissionKey)
            {
                _permissionKey = permissionKey;
            }
 
            public void OnAuthorization(
                AuthorizationFilterContext context)
            {
                var user = context.HttpContext.User;
 
                // 未登入 -> 導向登入頁
                if (!user.Identity.IsAuthenticated)
                {
                    context.Result = new RedirectToActionResult(
                        "Login", "Auth", new { area = "Admin" });
                    return;
                }
 
                // 無此權限 -> 導向 AccessDenied 頁
                if (!user.HasPermission(_permissionKey))
                {
                    context.Result = new RedirectToActionResult(
                        "AccessDenied", "Auth",
                        new { area = "Admin" });
                }
            }
        }

注意事項
------------------------------------------------------------
  * StoreStatusHelper 放在 ISpanShop.Common 確保各層都能引用
    不需要在每一層重複定義對應邏輯
  * updateStoreRow JS 函式需同步更新 .store-status-cell
    的 Badge 內容與 CSS class（從 AJAX 回傳的資料取得）
  * AJAX 回傳 store 物件需包含：
        storeStatus / storeStatusName / storeStatusBadge
  * IsBlacklisted 與 StoreStatus=3(停權) 是兩個獨立欄位
    封鎖（IsBlacklisted）= 平台操作
    停權（StoreStatus=3）= 營業狀態，語意不同請勿混用

PermissionKey 對應選單 速查表
------------------------------------------------------------
 
  PermissionKey      | 可見選單項目
  -------------------|------------------------------------
  product_manage     | 商品監管、分類管理
  store_manage       | 庫存監控、店家管理
  cashflow_manage    | 訂單管理、金流管理
  ticket_manage      | 內容與客服（工單功能）
  sensitive_manage   | 內容與客服（敏感詞功能）
  member_manage      | 會員管理
  report_view        | 首頁儀錶板
  超級管理員專用       | 登入紀錄、管理員權限設定
 
------------------------------------------------------------
注意事項
------------------------------------------------------------
 
  * STEP 1~5 為基礎建設，必須先完成才能進行後續步驟
  * STEP 6（Controller Attribute）是安全核心，
    選單隱藏只是 UX，Controller 保護才是真正防線
  * ContentModerationController 若工單與敏感詞在
    同一個 Controller，建議在 Action 層個別套用 Attribute
    而非整個 Controller 套用
  * IsSuperAdmin() 判斷依賴 Claim "AdminLevelId" == "1"，
    登入時務必確認此 Claim 有正確寫入
  * 若未來新增管理員身分，只需在資料庫
    AdminLevelPermissions 新增對應紀錄即可，
    程式碼不需修改
